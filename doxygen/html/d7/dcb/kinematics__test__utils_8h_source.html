144 inline void runJacobianTest(tesseract_kinematics::ForwardKinematics& kin,
145  const Eigen::VectorXd& jvals,
146  const std::string& link_name,
147  const Eigen::Vector3d& link_point,
148  const Eigen::Isometry3d& change_base)
149 {
150  Eigen::MatrixXd jacobian, numerical_jacobian;
151  Eigen::Isometry3d pose;
152 
153  jacobian.resize(6, kin.numJoints());
154  if (link_name.empty())
155  {
156  pose = kin.calcFwdKin(jvals);
157  jacobian = kin.calcJacobian(jvals);
158  }
159  else
160  {
161  pose = kin.calcFwdKin(jvals, link_name);
162  jacobian = kin.calcJacobian(jvals, link_name);
163  }
164  tesseract_kinematics::jacobianChangeBase(jacobian, change_base);
165  tesseract_kinematics::jacobianChangeRefPoint(jacobian, (change_base * pose).linear() * link_point);
166 
167  numerical_jacobian.resize(6, kin.numJoints());
168  if (link_name.empty())
169  tesseract_kinematics::numericalJacobian(
170  numerical_jacobian, change_base, kin, jvals, kin.getTipLinkName(), link_point);
171  else
172  tesseract_kinematics::numericalJacobian(numerical_jacobian, change_base, kin, jvals, link_name, link_point);
173 
174  for (int i = 0; i < 6; ++i)
175  for (int j = 0; j < static_cast<int>(kin.numJoints()); ++j)
176  EXPECT_NEAR(numerical_jacobian(i, j), jacobian(i, j), 1e-3);
177 }

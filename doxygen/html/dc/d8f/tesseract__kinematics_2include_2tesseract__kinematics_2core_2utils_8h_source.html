87 inline static void numericalJacobian(Eigen::Ref<Eigen::MatrixXd> jacobian,
88  const Eigen::Isometry3d& change_base,
89  const tesseract_kinematics::ForwardKinematics& kin,
90  const Eigen::Ref<const Eigen::VectorXd>& joint_values,
91  const std::string& link_name,
92  const Eigen::Ref<const Eigen::Vector3d>& link_point)
93 {
94  Eigen::VectorXd njvals;
95  double delta = 0.001;
96  Eigen::Isometry3d pose = kin.calcFwdKin(joint_values, link_name);
97  pose = change_base * pose;
98 
99  for (int i = 0; i < static_cast<int>(joint_values.size()); ++i)
100  {
101  njvals = joint_values;
102  njvals[i] += delta;
103  Eigen::Isometry3d updated_pose = kin.calcFwdKin(njvals, link_name);
104  updated_pose = change_base * updated_pose;
105 
106  Eigen::Vector3d temp = pose * link_point;
107  Eigen::Vector3d temp2 = updated_pose * link_point;
108  jacobian(0, i) = (temp2.x() - temp.x()) / delta;
109  jacobian(1, i) = (temp2.y() - temp.y()) / delta;
110  jacobian(2, i) = (temp2.z() - temp.z()) / delta;
111 
112  Eigen::AngleAxisd r12(pose.rotation().transpose() * updated_pose.rotation()); // rotation from p1 -> p2
113  double theta = r12.angle();
114  theta = copysign(fmod(fabs(theta), 2.0 * M_PI), theta);
115  if (theta < -M_PI)
116  theta = theta + 2. * M_PI;
117  if (theta > M_PI)
118  theta = theta - 2. * M_PI;
119  Eigen::VectorXd omega = (pose.rotation() * r12.axis() * theta) / delta;
120  jacobian(3, i) = omega(0);
121  jacobian(4, i) = omega(1);
122  jacobian(5, i) = omega(2);
123  }
124 }

134 inline static bool solvePInv(const Eigen::Ref<const Eigen::MatrixXd>& A,
135  const Eigen::Ref<const Eigen::VectorXd>& b,
136  Eigen::Ref<Eigen::VectorXd> x)
137 {
138  const double eps = 0.00001; // TODO: Turn into class member var
139  const double lambda = 0.01; // TODO: Turn into class member var
140 
141  if ((A.rows() == 0) || (A.cols() == 0))
142  {
143  CONSOLE_BRIDGE_logError("Empty matrices not supported in solvePinv()");
144  return false;
145  }
146 
147  if (A.rows() != b.size())
148  {
149  CONSOLE_BRIDGE_logError("Matrix size mismatch: A(%d, %d), b(%d)", A.rows(), A.cols(), b.size());
150  return false;
151  }
152 
153  // Calculate A+ (pseudoinverse of A) = V S+ U*, where U* is Hermition of U (just transpose if all values of U are
154  // real)
155  // in order to solve Ax=b -> x*=A+ b
156  Eigen::JacobiSVD<Eigen::MatrixXd> svd(A, Eigen::ComputeThinU | Eigen::ComputeThinV);
157  const Eigen::MatrixXd& U = svd.matrixU();
158  const Eigen::VectorXd& Sv = svd.singularValues();
159  const Eigen::MatrixXd& V = svd.matrixV();
160 
161  // calculate the reciprocal of Singular-Values
162  // damp inverse with lambda so that inverse doesn't oscillate near solution
163  long int nSv = Sv.size();
164  Eigen::VectorXd inv_Sv(nSv);
165  for (long int i = 0; i < nSv; ++i)
166  {
167  if (fabs(Sv(i)) > eps)
168  inv_Sv(i) = 1 / Sv(i);
169  else
170  inv_Sv(i) = Sv(i) / (Sv(i) * Sv(i) + lambda * lambda);
171  }
172  x = V * inv_Sv.asDiagonal() * U.transpose() * b;
173  return true;
174 }

185 inline static bool dampedPInv(const Eigen::Ref<const Eigen::MatrixXd>& A,
186  Eigen::Ref<Eigen::MatrixXd> P,
187  const double eps = 0.011,
188  const double lambda = 0.01)
189 {
190  if ((A.rows() == 0) || (A.cols() == 0))
191  {
192  CONSOLE_BRIDGE_logError("Empty matrices not supported in dampedPInv()");
193  return false;
194  }
195 
196  // Calculate A+ (pseudoinverse of A) = V S+ U*, where U* is Hermition of U (just transpose if all values of U are
197  // real)
198  // in order to solve Ax=b -> x*=A+ b
199  Eigen::JacobiSVD<Eigen::MatrixXd> svd(A, Eigen::ComputeThinU | Eigen::ComputeThinV);
200  const Eigen::MatrixXd& U = svd.matrixU();
201  const Eigen::VectorXd& Sv = svd.singularValues();
202  const Eigen::MatrixXd& V = svd.matrixV();
203 
204  // calculate the reciprocal of Singular-Values
205  // damp inverse with lambda so that inverse doesn't oscillate near solution
206  long int nSv = Sv.size();
207  Eigen::VectorXd inv_Sv(nSv);
208  for (long int i = 0; i < nSv; ++i)
209  {

211  inv_Sv(i) = 1 / Sv(i);
212  else
213  {
214  inv_Sv(i) = Sv(i) / (Sv(i) * Sv(i) + lambda * lambda);
215  }
216  }
